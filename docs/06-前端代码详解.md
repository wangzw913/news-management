## 一、前端启动流程

```
浏览器访问 http://localhost:5173
    │
    ▼
index.html 加载 → <script type="module" src="/src/main.js">
    │
    ▼
main.js 执行：
  1. 创建 Vue 应用实例
  2. 注册 Pinia（状态管理）+ 持久化插件
  3. 注册 Vue Router（路由）
  4. 注册 Element Plus（UI 组件库）+ 中文语言包
  5. 注册所有 Element Plus 图标
  6. 挂载到 <div id="app">
    │
    ▼
App.vue 渲染 → 只有一行 <router-view />
    │
    ▼
router/index.js 根据当前 URL 决定显示哪个组件
  /login → LoginView.vue
  /admin → AdminLayout.vue → 继续匹配子路由
  /      → PortalLayout.vue → 继续匹配子路由
```

### main.js —— 应用入口

```javascript
import { createApp } from 'vue'
import { createPinia } from 'pinia'
import piniaPluginPersistedstate from 'pinia-plugin-persistedstate'
import ElementPlus from 'element-plus'
import 'element-plus/dist/index.css'           // Element Plus 的 CSS
import zhCn from 'element-plus/es/locale/lang/zh-cn'  // 中文语言包
import * as ElementPlusIconsVue from '@element-plus/icons-vue'
import App from './App.vue'
import router from './router'
import './styles/global.scss'

const app = createApp(App)        // 1. 创建 Vue 应用

const pinia = createPinia()       // 2. 创建 Pinia 状态管理
pinia.use(piniaPluginPersistedstate)  // 3. 开启持久化（刷新不丢登录状态）
app.use(pinia)                    // 4. 注册 Pinia

app.use(router)                   // 5. 注册路由
app.use(ElementPlus, { locale: zhCn })  // 6. 注册 Element Plus（中文）

// 7. 注册所有 Element Plus 图标（全局可用，不需要每次 import）
for (const [key, component] of Object.entries(ElementPlusIconsVue)) {
  app.component(key, component)
}

app.mount('#app')                 // 8. 挂载到 <div id="app">
```

**为什么用 Pinia 而不是 Vuex？** Pinia 是 Vuex 的继任者，更简洁，TypeScript 支持更好，不需要 mutations。

### App.vue —— 根组件

```vue
<template>
  <router-view />  <!-- 只有这一行！根据 URL 动态切换显示的组件 -->
</template>

<script setup>
import { useUserStore } from '@/stores/user'

const userStore = useUserStore()
// 应用启动时：如果本地有 token，自动恢复登录状态
if (userStore.accessToken) {
  userStore.fetchUserInfo().catch(() => userStore.logout())
}
</script>
```

**App.vue 只做一件事**：用 `<router-view />` 占位，Vue Router 会根据 URL 把对应的组件渲染到这里。

启动时自动恢复登录状态的逻辑：如果 localStorage 里有 token → 调用 `/auth/me` 获取用户信息 → 成功则保持登录 → 失败则清空 token。

### 路由守卫 (beforeEach) 逐行解析

```javascript
router.beforeEach(async (to, from, next) => {
  // to：要去的页面   from：来自哪个页面   next：放行/跳转

  const userStore = useUserStore()

  // 第 1 步：公开页面直接放行
  if (to.meta.public) return next()

  // 第 2 步：未登录 → 强制去登录页
  if (!userStore.isLoggedIn) return next('/login')

  // 第 3 步：已登录但还没加载用户信息 → 先加载
  if (!userStore.username) {
    try {
      await userStore.fetchUserInfo()  // 调 /auth/me 接口
    } catch {
      userStore.logout()               // token 过期 → 登出
      return next('/login')
    }
  }

  // 第 4 步：已登录却访问登录页 → 按角色跳转
  if (to.path === '/login') {
    return next(userStore.isUser ? '/' : '/admin')
  }

  // 第 5 步：角色不匹配 → 403
  if (to.meta.roles && !to.meta.roles.includes(userStore.role)) {
    return next('/403')
  }

  // 第 6 步：一切正常 → 放行
  next()
})
```

**这段代码保证了**：未登录的人永远看不到需要登录的页面，角色不对的人永远看不到不该看的页面。

## 四、状态管理 —— stores/user.js

```javascript
import { defineStore } from 'pinia'

export const useUserStore = defineStore('user', () => {
  // ===== 状态（数据） =====
  const accessToken = ref('')      // JWT Access Token
  const refreshToken = ref('')     // JWT Refresh Token
  const userId = ref(null)         // 用户 ID
  const username = ref('')         // 用户名
  const role = ref('')             // 角色：admin/auditor/editor/user
  const roleName = ref('')         // 角色中文名：管理员/审核员/编辑/普通用户
  const avatar = ref('')           // 头像（预留）

  // ===== 计算属性 =====
  const isLoggedIn = computed(() => !!accessToken.value)  // 有 token = 已登录
  const isAdmin = computed(() => role.value === 'admin')
  const isAuditor = computed(() => role.value === 'auditor')
  const isEditor = computed(() => role.value === 'editor')
  const isUser = computed(() => role.value === 'user')

  // ===== 方法 =====
  async function login(credentials) {
    const data = await request.post('/auth/login', credentials)
    accessToken.value = data.accessToken
    refreshToken.value = data.refreshToken
    userId.value = data.userId
    username.value = data.username
    role.value = data.role
    roleName.value = data.roleName
    // 此时 Pinia 持久化插件自动把这些写入 localStorage
  }

  async function logout() {
    // 1. 通知后端登出（把 token 加入黑名单）
    if (refreshToken.value) {
      request.post('/auth/logout', { refreshToken: refreshToken.value }).catch(() => {})
    }
    // 2. 清空本地状态
    accessToken.value = ''
    refreshToken.value = ''
    userId.value = null
    username.value = ''
    role.value = ''
    roleName.value = ''
    // localStorage 也会被自动清空（因为持久化监听 ref 变化）
  }

  async function fetchUserInfo() {
    const data = await request.get('/auth/me')
    userId.value = data.userId
    username.value = data.username
    role.value = data.role
    roleName.value = data.roleName
    avatar.value = data.avatar
  }

  return {
    accessToken, refreshToken, userId, username, role, roleName, avatar,
    isLoggedIn, isAdmin, isAuditor, isEditor, isUser,
    login, register, fetchUserInfo, logout
  }
}, {
  // 持久化配置：这些字段存 localStorage，刷新页面不丢
  persist: {
    key: 'news-user',
    pick: ['accessToken', 'refreshToken', 'userId', 'username', 'role', 'roleName', 'avatar']
    // 注意：password 不在列表里，永远不会存本地！
  }
})
```

**任何时候在任何组件里**：
```javascript

const userStore = useUserStore()
console.log(userStore.username)    // 当前用户名
console.log(userStore.isAdmin)     // 是不是管理员
userStore.logout()                 // 登出
```

## 五、API 请求层 —— api/request.js

```javascript
import axios from 'axios'

const request = axios.create({
  baseURL: '/api/v1',     // 所有请求自动加这个前缀
  timeout: 15000          // 15 秒超时
})

// ===== 请求拦截器：发请求前做的事 =====
request.interceptors.request.use(
  config => {
    const userStore = useUserStore()
    if (userStore.accessToken) {
      // 自动带 Authorization 头
      config.headers.Authorization = `Bearer ${userStore.accessToken}`
    }
    return config
  },
  error => Promise.reject(error)
)

// ===== 响应拦截器：收到响应后做的事 =====
request.interceptors.response.use(
  response => {
    const { data } = response
    if (data.code === 200) {
      return data.data  // ← 自动解包！调用方直接拿到 data 里的内容
    }
    // 业务错误：弹提示
    ElMessage.error(data.message || '请求失败')
    return Promise.reject(new Error(data.message))
  },
  async error => {
    // HTTP 错误
    const { config, response } = error

    // Token 过期 → 自动刷新
    if (response?.status === 401 && !config._retry) {
      const userStore = useUserStore()
      if (!userStore.refreshToken) {
        userStore.logout()
        router.push('/login')
        return Promise.reject(error)
      }

      // 防止多个请求同时刷新 token（用队列）
      if (isRefreshing) {
        return new Promise((resolve) => {
          refreshQueue.push(resolve)
        }).then(() => {
          config.headers.Authorization = `Bearer ${userStore.accessToken}`
          return request(config)  // 重试原请求
        })
      }

      config._retry = true
      isRefreshing = true

      try {
        const { accessToken, refreshToken } = await refreshTokenApi()
        userStore.accessToken = accessToken
        userStore.refreshToken = refreshToken
        // 通知队列中等待的请求
        refreshQueue.forEach(resolve => resolve())
        refreshQueue = []
        // 重试原请求
        config.headers.Authorization = `Bearer ${accessToken}`
        return request(config)
      } catch (e) {
        refreshQueue.forEach(resolve => resolve()) // 即使失败也释放队列
        refreshQueue = []
        userStore.logout()
        router.push('/login')
        return Promise.reject(e)
      } finally {
        isRefreshing = false
      }
    }

    ElMessage.error(response?.data?.message || '网络错误')
    return Promise.reject(error)
  }
)
```

**关键设计：token 刷新队列**
如果有 3 个请求同时因为 token 过期返回 401，只有一个会真正去刷新 token，另外 2 个排队等待。刷新成功后，3 个请求都用新 token 重试。

```javascript
async function loadData() {
  const params = { page: page.value, size: size.value, ...filters }
  // 删除空值参数（不传空字符串给后端）
  Object.keys(params).forEach(k => { if (!params[k]) delete params[k] })
  const data = await request.get('/news', { params })
  list.value = data.records
  total.value = data.total
}
```

## 八、前端面试要点

### 为什么用 Vue 3 Composition API？

比 Options API 更灵活：逻辑可以抽成 composables 复用，不需要 this，TypeScript 支持更好。

### ref 和 reactive 的区别？

- `ref`：包装基本类型（字符串、数字），需要 `.value` 访问
- `reactive`：包装对象，直接访问属性，不需要 `.value`
- 我的习惯：单个值用 ref，表单对象用 reactive

### v-if 和 v-show 的区别？

- `v-if`：条件为假时 DOM 元素不存在（销毁和重建），切换开销大
- `v-show`：条件为假时 DOM 还在，只是 `display:none`，切换开销小
- 审核队列用 `v-show` 切换全文：频繁切换，用 v-show 性能更好

### computed 和 method 的区别？

- `computed`：有缓存，依赖不变不重新计算
- `method`：每次调用都重新执行
- 示例：`isAdmin` 用 computed，因为 role 不变就不用重新判断

### 组件间怎么通信？

- 父子：props（父→子）+ emit（子→父）
- 跨组件：Pinia store（全局状态）
- 路由参数：`route.params.id`（页面间传参）
